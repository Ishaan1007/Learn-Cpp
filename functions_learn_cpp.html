Functions in C++

A function is a self-contained block of code that performs a specific task. Functions help in code reusability, readability, and modular programming.

Types of Functions:

  • Library Functions: Predefined (like sqrt(), pow(), strlen()).
  • User-defined Functions: Written by the programmer.

Components of a Function:

  • Declaration: Tells the compiler about the function (prototype).
  • Definition: Actual body of the function.
  • Call: Invoking the function.

Categories:

  • Functions with arguments and return value.
  • Functions with arguments and no return value.
  • Functions with no arguments and return value.
  • Functions with no arguments and no return value.

Examples

1. Function to Add Two Numbers

#include <iostream>
using namespace std;

int add(int a, int b) {
    return a + b;
}

int main() {
    int x = 5, y = 10;
    cout << "Sum = " << add(x, y);
    return 0;
}

2. Function with No Arguments, No Return Value

#include <iostream>
using namespace std;

void greet() {
    cout << "Hello, Welcome to C++!";
}

int main() {
    greet();
    return 0;
}

3. Recursive Function: Factorial

#include <iostream>
using namespace std;

int factorial(int n) {
    if(n == 0) return 1;
    return n * factorial(n - 1);
}

int main() {
    int num = 5;
    cout << "Factorial = " << factorial(num);
    return 0;
}

4. Menu-Driven Calculator

#include <iostream>
using namespace std;

int add(int a, int b) { return a+b; }
int sub(int a, int b) { return a-b; }
int mul(int a, int b) { return a*b; }
float divi(int a, int b) { return (float)a/b; }

int main() {
    int choice,a,b;
    cout << "1.Add 2.Sub 3.Mul 4.Div\n";
    cin >> choice >> a >> b;
    switch(choice){
        case 1: cout << add(a,b); break;
        case 2: cout << sub(a,b); break;
        case 3: cout << mul(a,b); break;
        case 4: cout << divi(a,b); break;
        default: cout << "Invalid";
    }
}

5. Student Result System

#include <iostream>
using namespace std;

float percentage(int m1,int m2,int m3,int m4,int m5){
    return (m1+m2+m3+m4+m5)/5.0;
}

char grade(float p){
    if(p>=90) return 'A';
    else if(p>=75) return 'B';
    else if(p>=50) return 'C';
    else return 'F';
}

int main(){
    int m1,m2,m3,m4,m5;
    cout << "Enter 5 subject marks: ";
    cin >> m1 >> m2 >> m3 >> m4 >> m5;
    float p = percentage(m1,m2,m3,m4,m5);
    cout << "Percentage: " << p << "%\nGrade: " << grade(p);
}

7. Password Validator

#include <iostream>
#include <cctype>
using namespace std;

bool isValidPassword(string pwd) {
    bool upper=false, lower=false, digit=false, special=false;
    for(char c: pwd){
        if(isupper(c)) upper=true;
        else if(islower(c)) lower=true;
        else if(isdigit(c)) digit=true;
        else special=true;
    }
    return (upper && lower && digit && special && pwd.size()>=8);
}

int main(){
    string pass;
    cout << "Enter password: ";
    cin >> pass;
    if(isValidPassword(pass))
        cout << "Strong Password ✅";
    else
        cout << "Weak Password ❌";
}

8. Calculator (with recursion for power a^b)

#include <iostream>
using namespace std;

int add(int a, int b) { return a+b; }
int sub(int a, int b) { return a-b; }
int mul(int a, int b) { return a*b; }
float divi(int a, int b) { return (float)a/b; }
int power(int base, int exp) {
    if(exp == 0) return 1;
    return base * power(base, exp-1);   // recursion
}

int main() {
    int choice,a,b;
    cout << "1.Add 2.Sub 3.Mul 4.Div 5.Power\n";
    cin >> choice >> a >> b;
    switch(choice){
        case 1: cout << add(a,b); break;
        case 2: cout << sub(a,b); break;
        case 3: cout << mul(a,b); break;
        case 4: cout << divi(a,b); break;
        case 5: cout << power(a,b); break;
        default: cout << "Invalid Choice";
    }
}

9. Bank Account Simulation (with PIN verification)

#include <iostream>
using namespace std;

int balance = 2000;
int pin = 1234;

bool verifyPin(int entered) {
    return entered == pin;
}

void deposit(int amt){ balance += amt; }
void withdraw(int amt){ 
    if(amt > balance) cout << "Insufficient balance\n";
    else balance -= amt; 
}
void showBalance(){ cout << "Balance: " << balance << endl; }

int main(){
    int enteredPin;
    cout << "Enter PIN: ";
    cin >> enteredPin;

    if(!verifyPin(enteredPin)){
        cout << "❌ Wrong PIN. Access Denied.";
        return 0;
    }

    int choice, amt;
    cout << "1.Deposit 2.Withdraw 3.Check Balance\n";
    cin >> choice;

    switch(choice){
        case 1: cout << "Amount: "; cin >> amt; deposit(amt); break;
        case 2: cout << "Amount: "; cin >> amt; withdraw(amt); break;
        case 3: showBalance(); break;
        default: cout << "Invalid Option";
    }

    showBalance();
}
