Operators

Arithmetic, relational, logical, bitwise, assignment and more.

Arithmetic

int a = 10, b = 3;
int sum = a + b;    // +   -> 13
int diff = a - b;   // -   -> 7
int prod = a * b;   // *   -> 30
int divi = a / b;   // /   -> 3 (integer division)
int mod = a % b;    // %   -> 1

a++; ++a; // increment (postfix & prefix)
b--; --b; // decrement

Relational / Comparison

bool eq = (a == b); // equal
bool ne = (a != b);      // not equal
bool gt = (a > b);
bool lt = (a < b);
bool ge = (a >= b);
bool le = (a <= b);

Logical

bool res = (a > 0) && (b > 0); // logical AND
bool any = (a > 0) || (b > 0);        // logical OR
bool notx = !(a > 0);                   // logical NOT

Bitwise

int x = 5;      // 0101
int y = 3;      // 0011
int anded = x & y;  // &  -> 0001 -> 1
int ored  = x | y;      // |   -> 0111 -> 7
int xored = x ^ y;      // ^   -> 0110 -> 6
int shl   = x << 1; // <<  -> 1010 -> 10
int shr   = x >> 1; // >>  -> 0010 -> 2
int neg   = ~x;         // ~   -> bitwise NOT

Assignment & Compound

int v = 10;
v += 5;  // v = v + 5
v -= 2;  // v = v - 2
v *= 3;  // v = v * 3
v /= 2;  // v = v / 2
v %= 4;  // v = v % 4

Ternary

int sign = (x >= 0) ? 1 : -1;

Control Flow — if, else-if, switch

// if-else
if (x > 0) {
    cout << "positive";
} else if (x == 0) {
    cout << "zero";
} else {
    cout << "negative";
}

// switch (integral only)
int day = 2;
switch (day) {
  case 1: cout << "Mon"; break;
  case 2: cout << "Tue"; break;

Loops — for / while / do-while

Classic and range-based loops.

For loop

for (int i = 0; i < 5; ++i) {
    cout << i << ' ';
}

Range-based for (C++11+)

std::vector<int> v = {1,2,3};
for (int &x : v) {     // iterate by reference
    cout << x << ' ';
}

While

int n = 5;
while (n > 0) {
    cout << n-- << ' ';
}

Do-while

int i = 0;
do {
    cout << i << ' ';
    ++i;
} while (i < 3);

Nested loops example

for (int r=0; r<3; ++r){
    for (int c=0; c<3; ++c){
        cout << r << ',' << c << ' ';
    }
}

Inheritance & Polymorphism

Base/derived classes, virtual functions and overriding.

class Base {
public:
    virtual void speak(){ cout << "base\n"; }
    virtual ~Base() = default;
};
class Derived : public Base {
public:
    void speak() override { cout << "derived\n"; }
};

Base* b = new Derived();
b->speak(); // calls Derived::speak
delete b;
