Arrays in C++

An array is a collection of elements of the same type stored in contiguous memory locations. Each element can be accessed using its index (starting from 0).

Types of Arrays:

  • One-dimensional (1D): Linear collection of elements.
  • Two-dimensional (2D): Represented as rows and columns (matrices).
  • Multi-dimensional: Arrays with more than two dimensions.

Strings in C++

A string is a sequence of characters terminated by a null character ('\0').

Ways to Handle Strings:

  • C-style strings using character arrays.
  • String class in C++ (provides advanced functionality).

Examples

1. One-Dimensional Array

#include <iostream>
using namespace std;

int main() {
    int marks[5] = {90, 85, 78, 92, 88};
    for(int i=0; i<5; i++) {
        cout << "Marks[" << i << "] = " << marks[i] << endl;
    }
    return 0;
}

2. Two-Dimensional Array (Matrix)

#include <iostream>
using namespace std;

int main() {
    int mat[2][3] = {{1,2,3},{4,5,6}};
    for(int i=0; i<2; i++) {
        for(int j=0; j<3; j++) {
            cout << mat[i][j] << " ";
        }
        cout << endl;
    }
    return 0;
}

5. Find Secind Largest Element in Array

#include <iostream>
using namespace std;

int main() {
    int arr[] = {10, 45, 67, 89, 12, 67};
    int n = 6;
    int first = -1e9, second = -1e9;

    for(int i=0;i<n;i++){
        if(arr[i] > first){
            second = first;
            first = arr[i];
        } 
        else if(arr[i] > second && arr[i] != first){
            second = arr[i];
        }
    }
    cout << "Second Largest = " << second;
}

6. Rotate Array by k Positions

#include <iostream>
using namespace std;

void rotateRight(int arr[], int n, int k){
    k = k % n;
    int temp[n];
    for(int i=0;i<n;i++){
        temp[(i+k)%n] = arr[i];
    }
    for(int i=0;i<n;i++) arr[i] = temp[i];
}

int main(){
    int arr[] = {1,2,3,4,5,6};
    int n=6, k=2;
    rotateRight(arr,n,k);
    for(int x: arr) cout << x << " ";
}

7. Matrix Multiplication (2D Arrays)

#include <iostream>
using namespace std;

int main(){
    int A[2][2]={{1,2},{3,4}};
    int B[2][2]={{5,6},{7,8}};
    int C[2][2]={0};

    for(int i=0;i<2;i++){
        for(int j=0;j<2;j++){
            for(int k=0;k<2;k++){
                C[i][j] += A[i][k]*B[k][j];
            }
        }
    }

    cout << "Result Matrix:\n";
    for(int i=0;i<2;i++){
        for(int j=0;j<2;j++){
            cout << C[i][j] << " ";
        }
        cout << endl;
    }
}

8. Find Duplicate Elements in Array

#include <iostream>
using namespace std;

int main(){
    int arr[] = {4, 5, 6, 4, 7, 5, 8};
    int n = 7;

    cout << "Duplicate Elements: ";
    for(int i=0;i<n;i++){
        for(int j=i+1;j<n;j++){
            if(arr[i]==arr[j]){
                cout << arr[i] << " ";
                break;
            }
        }
    }
}

9. Pascals Triangle using Arrays

#include <iostream>
using namespace std;

int main(){
    int n=5;
    int arr[10][10]={0};

    for(int i=0;i<n;i++){
        arr[i][0]=arr[i][i]=1;
        for(int j=1;j<i;j++)
            arr[i][j] = arr[i-1][j-1]+arr[i-1][j];
    }

    for(int i=0;i<n;i++){
        for(int j=0;j<=i;j++)
            cout << arr[i][j] << " ";
        cout << endl;
    }
}
